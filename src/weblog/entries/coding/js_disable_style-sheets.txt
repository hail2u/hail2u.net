スタイルシートを無効にするJavaScript
<pre><code class="language-js">&lt;script type=&quot;text/javascript&quot; defer&gt;
  var pWin = external.menuArguments;
  var docs = pWin.document;
  for(i = 0; i &lt; docs.styleSheets.length; i++){
    docs.styleSheets.item(i).disabled = true;
  }
&lt;/script&gt;
