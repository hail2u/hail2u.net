Hide line-number when selecting pre text

pre {
  counter-reset: lnum;
}

pre .lnum::before {
  margin-right: 1em;
  display: inline-block;
  width: 3em;
  content: counter(lnum) ":";
  counter-increment: lnum;
  color: #999;
  text-align: right;
}

pre:active .lnum::before {
  visibility: hidden;
}
