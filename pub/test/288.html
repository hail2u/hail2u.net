C&P Safe pre Line Number

JavaScript

$(function () {
  $("pre").html(function (i, h) {
    return h.replace(/^(.*?)$/gm, "$1");
  });
});

CSS

pre {
  counter-reset: lnum;
}

pre .lnum::before {
  margin-right: 1em;
  display: inline-block;
  width: 3em;
  content: counter(lnum) ":";
  counter-increment: lnum;
  color: #999;
  text-align: right;
}
